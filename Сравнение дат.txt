#include <iostream>

using namespace std;

bool compareDates()
{
    int intDate1[3] = { 2019, 9, 14 };
    int intDate2[3] = { 2016, 3, 11 };

    int i = 0;

    if (intDate1[0] == intDate2[0] && intDate1[1] == intDate2[1] && intDate1[2] == intDate2[2])
    {
        return true;
    }

    while (i < 3)
    {
        if (intDate1[i] > intDate2[i])
        {
            return false;
        }
        else if (intDate1[i] < intDate2[i])
        {
            return true;
        }
            i++;
    }
}


bool compareDates2(char* date1, char* date2)
{
    int result = strcmp(date1, date2);

    if (result > 0)
    {
        return false;
    }

    else
    {
        return true;
    }
}

int main()
{
    char date1 [] = "2019.09.14";
    char date2 [] = "2016.03.11";

    cout << compareDates() << endl;
    cout << compareDates2(date1, date2) << endl;

    return 0;
}
